-- Função auxiliar para adicionar botões que carregam scripts

        section:NewButton(script[1], "Clique para executar", function()
            loadstring(game:HttpGet(script[2]))()
        end)
    end
end

for _, script in ipairs(outros) do
    if script[2] == "local" then
        SectionOutros:NewButton(script[1], "Clique para executar", function()
            if script[1] == "Rejoin" then
                game:GetService("TeleportService"):TeleportToPlaceInstance(game.PlaceId, game.JobId, game.Players.LocalPlayer)
            elseif script[1] == "Server Hop" then
                local TeleportService = game:GetService("TeleportService")
                local HttpService = game:GetService("HttpService")
                local PlaceId = game.PlaceId

                local servers = HttpService:JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/" .. PlaceId .. "/servers/Public?sortOrder=Desc&limit=100")).data
                for _, server in pairs(servers) do
                    if server.playing < server.maxPlayers then
                        TeleportService:TeleportToPlaceInstance(PlaceId, server.id)
                        break
                    end
                end
            end
        end)
    else
        SectionOutros:NewButton(script[1], "Clique para executar", function()
            loadstring(game:HttpGet(script[2]))()
        end)
    end
end
