local loadedScripts = {}

for key, url in pairs(scriptsURLs) do
    coroutine.wrap(function()
        local success, code = pcall(function()
            return game:HttpGet(url)
        end)
        if success and code then
            local func = loadstring(code)
            loadedScripts[key] = func
        else
            warn("Falha ao carregar "..key.." -> "..tostring(code))
        end
    end)()
end

local function enableXray()
    xrayParts = {}
    for _, v in pairs(workspace:GetDescendants()) do
        if v:IsA("BasePart") and v.Transparency < 1 and v.CanCollide then
            table.insert(xrayParts, {part = v, transparency = v.Transparency})
            v.Transparency = 0.7
        end
    end
end

local function disableXray()
    for _, v in pairs(xrayParts) do
        if v.part and v.part.Parent then
            v.part.Transparency = v.transparency
        end
    end
    xrayParts = {}
end

local function FindPlayer(name)
    name = name:lower()
    for _, player in ipairs(Players:GetPlayers()) do
        if player.Name:lower():find(name) or player.DisplayName:lower():find(name) then
            return player
        end
    end
end

LocalPlayer.Chatted:Connect(function(msg)
    if msg:sub(1, #prefix):lower() ~= prefix then return end
    local args = msg:sub(#prefix + 1):split(" ")
    local cmd = args[1]:lower()

    if not character or not character.Parent then return end
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    local hrp = character:FindFirstChild("HumanoidRootPart")
    if not humanoid or not hrp then return end

    coroutine.wrap(function()
        if cmd == "tp" and args[2] then
            local destino = args[2]:lower()
            local pos = teleportAliases[destino]
            if pos then
                hrp.CFrame = CFrame.new(pos)
            else
                local alvo = FindPlayer(destino)
                if alvo and alvo.Character and alvo.Character:FindFirstChild("HumanoidRootPart") then
                    hrp.CFrame = alvo.Character.HumanoidRootPart.CFrame
                end
            end

        elseif cmd == "tpsofa" then
            hrp.CFrame = CFrame.new(-83, 21, -131)
            humanoid:ChangeState(Enum.HumanoidStateType.Seated)

        elseif cmd == "speed" and args[2] then
            humanoid.WalkSpeed = tonumber(args[2]) or 16

        elseif cmd == "resetspeed" then
            humanoid.WalkSpeed = 16

        elseif cmd == "jumppower" and args[2] then
            humanoid.JumpPower = tonumber(args[2]) or 50

        elseif cmd == "resetjp" then
            humanoid.JumpPower = 50

        elseif cmd == "speedsupreme" then
            local f = loadedScripts.speedsupreme
            if f then f() end

        elseif cmd == "view" and args[2] then
            local target = FindPlayer(args[2])
            if target and target.Character then
                local hum = target.Character:FindFirstChildOfClass("Humanoid")
                if hum then
                    Camera.CameraSubject = hum
                end
            end

        elseif cmd == "unview" then
            Camera.CameraSubject = humanoid

        elseif cmd == "xray" then
            enableXray()

        elseif cmd == "unxray" then
            disableXray()

        elseif cmd == "coolkid" then
            local f = loadedScripts.coolkid
            if f then f() end

        elseif cmd == "portalgun" then
            local f = loadedScripts.portalgun
            if f then f() end

        elseif cmd == "void" then
            local f = loadedScripts.void
            if f then f() end

        elseif cmd == "gravitygun" then
            local f = loadedScripts.gravitygun
            if f then f() end

        elseif cmd == "angelhub" then

            if f then f() end

            local f = loadedScripts.gigante
            if f then f() end

    end)()
